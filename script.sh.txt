#!/bin/bash

sudo apt update -y
sudo apt install apache2 php php-mysql mysql-client -y
sudo systemctl start apache2
cd /var/www/html/
sudo rm -r *

cat <<EOF > /var/www/html/index.php
<h2>Employee Form</h2>
<form action="insert.php" method="POST">
Name:<input type="text" name="name"><br>
Email:<input type="text" name="email"><br>
Department:<input type="text" name="department"><br>
<input type="submit">
</form>
<a href="view.php">View Employees</a>
EOF

cat <<EOF > /var/www/html/insert.php
<?php
\$conn = new mysqli("database-1.c8h68ggam6us.us-east-1.rds.amazonaws.com","admin","password123","employeedb");
\$name=\$_POST['name'];
\$email=\$_POST['email'];
\$dept=\$_POST['department'];
\$conn->query("INSERT INTO employees(name,email,department)
VALUES('\$name','\$email','\$dept')");
echo "Employee Added";
?>
EOF

cat <<EOF > /var/www/html/view.php
<?php
\$conn = new mysqli("database-1.c8h68ggam6us.us-east-1.rds.amazonaws.com","admin","password123","employeedb");
\$result=\$conn->query("SELECT * FROM employees");
while(\$row=\$result->fetch_assoc()){
echo \$row['name']." ".\$row['email']." ".\$row['department']."<br>";
}
?>
EOF

systemctl restart apache2
